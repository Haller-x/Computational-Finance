# Simula√ß√µes
n_sim <- 100000
total_sim <- matrix(0, nrow = 256, ncol = n_sim)
p_inicial <- last(BVSP$Fechamento)      

for(j in 1:n_sim){
  aux <- 2
  p <- p_inicial # Pre√ßo inicial
  preco <- c(p) # Vetor para acumular os precos ao longo do tempo
  dist <- rnorm(255,0,1) # Distribui√ß√£o normal, 255 valores, com m√©dia 0 e desvio p. 1
  for(i in dist)
  {
    P = p + p*(u/255 + sd/sqrt(255)*i)
    preco[aux] <- P                        
    p = P                                 
    aux = aux + 1
  }
  total_sim[,j] <- preco
}
